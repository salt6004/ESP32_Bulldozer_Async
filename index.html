#include #include #include #include #include #include "soc/soc.h" // ★追加: ブラウンアウト対策用 #include "soc/rtc_cntl_reg.h" // ★追加: ブラウンアウト対策用 // ============================================================================= // ピン設定 // ============================================================================= const int motorA_IN1 = 27; const unsigned long FAILSAFE_TIMEOUT = 3000; // ============================================================================= // BLE設定 // ============================================================================= BLEServer *pServer = NULL; BLECharacteristic *pRxCharacteristic = NULL; BLECharacteristic *pTxCharacteristic = NULL; // スマホへの通知用 bool deviceConnected = false; bool oldDeviceConnected = false; #define SERVICE_UUID "6E400001-B5A3-F393-E0A9-E50E24DCCA9E" #define CHARACTERISTIC_UUID_RX "6E400002-B5A3-F393-E0A9-E50E24DCCA9E" #define CHARACTERISTIC_UUID_TX "6E400003-B5A3-F393-E0A9-E50E24DCCA9E" // ============================================================================= // タスク間通信用キュー // ============================================================================= struct MotorCommand { char direction[20]; int speed; }; QueueHandle_t motorCmdQueue; // ============================================================================= // DRV8835 IN/INモード 低レベル出力 // ============================================================================= void hardCoastAll() { ledcWrite(motorA_IN1, 0); ledcWrite(motorA_IN2, 0); ledcWrite(motorB_IN1, 0); ledcWrite(motorB_IN2, 0); } void setMotorA(bool forward, int speed) { if (forward) { ledcWrite(motorA_IN2, 0); ledcWrite(motorA_IN1, speed); } else { ledcWrite(motorA_IN1, 0); ledcWrite(motorA_IN2, speed); } } void setMotorB(bool forward, int speed) { if (forward) { ledcWrite(motorB_IN1, 0); ledcWrite(motorB_IN2, speed); } else { ledcWrite(motorB_IN2, 0); ledcWrite(motorB_IN1, speed); } } // ============================================================================= // ソフトスタート付きモーター駆動 // ============================================================================= void driveWithRamp(bool dirA, bool dirB, const char* expectedMotion) { int spd = currentSpeed; int tgt = (int)targetSpeed; while (spd < tgt) { MotorCommand nextCmd; // 次に待っているコマンドを確認 if (xQueuePeek(motorCmdQueue, &nextCmd, 0) == pdTRUE) { // 同じ方向で同じ速度なら無視して加速継続 if (strcmp(nextCmd.direction, expectedMotion) == 0 && nextCmd.speed == tgt) { xQueueReceive(motorCmdQueue, &nextCmd, 0); } else { // 違うコマンド（stop等）が来た場合は即座に加速中断 currentSpeed = spd; return; } } spd = min(spd + RAMP_STEP_SIZE, tgt); setMotorA(dirA, spd); setMotorB(dirB, spd); vTaskDelay(pdMS_TO_TICKS(RAMP_INTERVAL_MS)); } setMotorA(dirA, tgt); setMotorB(dirB, tgt); currentSpeed = tgt; } // ============================================================================= // モーター制御コマンドのディスパッチ // ============================================================================= void dispatchMotorCommand(const char* direction, int speed) { targetSpeed = (int)constrain(speed, 0, 255); String dirStr = String(direction); // ブロックUIからの文字(F, B, L, R, S, W)を内部の制御名に変換 if (dirStr == "F") dirStr = "up"; else if (dirStr == "B") dirStr = "down"; else if (dirStr == "L") dirStr = "left"; else if (dirStr == "R") dirStr = "right"; else if (dirStr == "S") dirStr = "stop"; Serial.printf("[Motor] cmd=%s spd=%d\n", dirStr.c_str(), (int)targetSpeed); if (dirStr == "stop") { hardCoastAll(); currentSpeed = 0; currentMotion = "stop"; return; } if (dirStr == "W") { // W(まつ)の場合はモーターを止めて状態だけWにする hardCoastAll(); currentSpeed = 0; currentMotion = "W"; return; } // 方向切替時のデッドタイム bool needDeadtime = (currentMotion != dirStr); if (needDeadtime && currentMotion != "stop" && currentMotion != "W") { hardCoastAll(); vTaskDelay(pdMS_TO_TICKS(DEADTIME_MS)); } currentMotion = dirStr; if (dirStr == "up") { driveWithRamp(true, true, "up"); } else if (dirStr == "down") { driveWithRamp(false, false, "down"); } else if (dirStr == "left") { driveWithRamp(true, false, "left"); } else if (dirStr == "right") { driveWithRamp(false, true, "right"); } } // ============================================================================= // BLEコールバック (Core0) // ============================================================================= class MyServerCallbacks : public BLEServerCallbacks { void onConnect(BLEServer* pServer) override { deviceConnected = true; lastCommunicationTime = millis(); Serial.println("[BLE] Connected"); } void onDisconnect(BLEServer* pServer) override { deviceConnected = false; Serial.println("[BLE] Disconnected"); // ※通信が切れてもプログラム実行中の場合はストップさせない } }; class MyCallbacks : public BLECharacteristicCallbacks { void onWrite(BLECharacteristic *pCharacteristic) override { String rxValue = pCharacteristic->getValue().c_str(); if (rxValue.length() == 0) return; lastCommunicationTime = millis(); if (rxValue == "ping") return; int sep1 = rxValue.indexOf(':'); if (sep1 == -1) return; String cmd = rxValue.substring(0, sep1); String remain = rxValue.substring(sep1 + 1); // プログラム追加 "ADD:F:200:1000" if (cmd == "ADD") { int sep2 = remain.indexOf(':'); int sep3 = remain.lastIndexOf(':'); if (sep2 != -1 && sep3 != -1 && sep2 != sep3) { String dir = remain.substring(0, sep2); int spd = remain.substring(sep2 + 1, sep3).toInt(); int ms = remain.substring(sep3 + 1).toInt(); if (programCount < 35) { strncpy(programList[programCount].direction, dir.c_str(), sizeof(programList[programCount].direction) - 1); programList[programCount].speed = spd; programList[programCount].durationMs = ms; programCount++; Serial.printf("Added to Prog[%d]: %s %d %dms\n", programCount-1, dir.c_str(), spd, ms); } } } // プログラムクリア "CLR:0" else if (cmd == "CLR") { programCount = 0; Serial.println("Program Cleared."); } // プログラム実行 "RUN:0" else if (cmd == "RUN") { MotorCommand mcmd; strcpy(mcmd.direction, "RUN"); mcmd.speed = 0; xQueueSend(motorCmdQueue, &mcmd, pdMS_TO_TICKS(10)); } // 通常コマンド (手動や緊急停止など) "S:0" else { MotorCommand mcmd; strncpy(mcmd.direction, cmd.c_str(), sizeof(mcmd.direction) - 1); mcmd.speed = remain.toInt(); if (xQueueSend(motorCmdQueue, &mcmd, pdMS_TO_TICKS(10)) == errQUEUE_FULL) { MotorCommand dummy; xQueueReceive(motorCmdQueue, &dummy, 0); xQueueSend(motorCmdQueue, &mcmd, 0); } } } }; // ============================================================================= // モータータスク (Core1) // ============================================================================= void motorTask(void* pvParameters) { MotorCommand cmd; for (;;) { // 1. キューから命令を受け取る if (xQueueReceive(motorCmdQueue, &cmd, pdMS_TO_TICKS(10)) == pdTRUE) { if (strcmp(cmd.direction, "RUN") == 0) { // プログラム実行開始 if (programCount > 0) { isExecutingProgram = true; currentProgramIndex = 0; stepStartTime = millis(); Serial.println("[Program] Execution Started!"); dispatchMotorCommand(programList[0].direction, programList[0].speed); } } else if (strcmp(cmd.direction, "stop") == 0 || strcmp(cmd.direction, "S") == 0) { // ストップが送られた場合はプログラム実行を強制中断 isExecutingProgram = false; dispatchMotorCommand("stop", 0); } else { // その他の手動コマンド（手動操作でプログラム割り込み） isExecutingProgram = false; dispatchMotorCommand(cmd.direction, cmd.speed); } } // 2. 自律プログラム実行中の処理 if (isExecutingProgram) { unsigned long now = millis(); if (now - stepStartTime >= (unsigned long)programList[currentProgramIndex].durationMs) { // 指定時間が経過したので次のステップへ currentProgramIndex++; if (currentProgramIndex >= programCount) { // 全ステップ完了 isExecutingProgram = false; dispatchMotorCommand("stop", 0); Serial.println("[Program] All Steps Finished!"); // ★スマホ側に完了を通知する if (deviceConnected && pTxCharacteristic != NULL) { pTxCharacteristic->setValue("FINISH"); pTxCharacteristic->notify(); } } else { // 次の命令を実行 stepStartTime = millis(); dispatchMotorCommand(programList[currentProgramIndex].direction, programList[currentProgramIndex].speed); } } } // 3. プログラム実行中でない時のみフェイルセーフを監視 else { if (deviceConnected && currentMotion != "stop" && currentMotion != "W") { if (millis() - lastCommunicationTime > FAILSAFE_TIMEOUT) { Serial.println("[Failsafe] Timeout -> stop"); dispatchMotorCommand("stop", 0); } } } } } // ============================================================================= // setup // ============================================================================= void setup() { // ★追加: 電圧降下によるESP32の勝手な再起動(ブラウンアウト)を無効にする WRITE_PERI_REG(RTC_CNTL_BROWN_OUT_REG, 0); Serial.begin(115200); ledcAttach(motorA_IN1, PWM_FREQ, PWM_RESOLUTION); ledcAttach(motorA_IN2, PWM_FREQ, PWM_RESOLUTION); ledcAttach(motorB_IN1, PWM_FREQ, PWM_RESOLUTION); ledcAttach(motorB_IN2, PWM_FREQ, PWM_RESOLUTION); hardCoastAll(); motorCmdQueue = xQueueCreate(5, sizeof(MotorCommand)); BLEDevice::init("ESP32-Bulldozer"); BLEDevice::setPower(ESP_PWR_LVL_N9); pServer = BLEDevice::createServer(); pServer->setCallbacks(new MyServerCallbacks()); BLEService *pService = pServer->createService(SERVICE_UUID); pRxCharacteristic = pService->createCharacteristic( CHARACTERISTIC_UUID_RX, BLECharacteristic::PROPERTY_WRITE | BLECharacteristic::PROPERTY_WRITE_NR ); pRxCharacteristic->setCallbacks(new MyCallbacks()); // ★スマホへの通知用Characteristicを初期化 pTxCharacteristic = pService->createCharacteristic( CHARACTERISTIC_UUID_TX, BLECharacteristic::PROPERTY_NOTIFY ); pTxCharacteristic->addDescriptor(new BLE2902()); pService->start(); BLEAdvertising *pAdvertising = BLEDevice::getAdvertising(); pAdvertising->addServiceUUID(SERVICE_UUID); pAdvertising->setScanResponse(true); pAdvertising->setMinPreferred(0x06); pAdvertising->setMaxPreferred(0x12); BLEDevice::startAdvertising(); xTaskCreatePinnedToCore(motorTask, "MotorTask", 4096, NULL, 2, NULL, 1); Serial.println("[System] Ready. BLE=Core0 Motor=Core1"); } // ============================================================================= // loop (Core0) // ============================================================================= void loop() { if (!deviceConnected && oldDeviceConnected) { vTaskDelay(pdMS_TO_TICKS(10)); pServer->startAdvertising(); Serial.println("[BLE] Restarting advertising instantly"); oldDeviceConnected = false; } if (deviceConnected && !oldDeviceConnected) { oldDeviceConnected = true; } vTaskDelay(pdMS_TO_TICKS(10)); }
